# 计算公式

function getHouseRate(originRate = 4.9, discount = 1) {
    const LPR = 4.8;
    let rate = originRate * discount;
    let gap = rate - LPR;
    console.log("固定加点数：" + gap.toFixed(3));
    return function(newLPR = 4.8) {
        return Number((newLPR + gap).toFixed(3));
    }
}
// 例子1：房贷是1.1倍，也就是4.9 * 1.1 = 5.39%
var f = getHouseRate(4.9, 1.1); // 固定加点数：0.59
// 当LPR是4.8时候的利率：
f(4.8); // 5.39
// 当LPR是4.5时候的利率：
f(4.5); // 5.09
// 例子2：房贷是9折，也就是4.9% * 0.9 = 4.41%
var f = getHouseRate(4.9, 0.9); // 固定加点数：-0.39
// 当LPR是4.8时候的利率：
f(4.8); // 4.41
// 当LPR是4.5时候的利率：
f(4.5); // 4.11
